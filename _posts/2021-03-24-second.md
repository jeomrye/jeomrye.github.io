# 두번째 

# <script>
        //함수의 다양한 정의 방법

        //1. 함수 선언문
        function numbering() {
            i = 0;
            while (i < 10) {
                document.write(i);
                i += 1;
            }
            document.write("<br/>");
        }
        numbering();

        //2. 함수 표현식
        numbering = function() {
            i = 0;
            while (i < 10) {
                document.write(i);
                i += 1;
            }
        }
        numbering();
		
		두개의 차이는
		함수 선언문은 호출을 어디서든 할 수 있다
		함수 표현식은 함수 밑에 호출을 할 수 있다
